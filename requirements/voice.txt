# Voice input/output dependencies.
# Required for --audio mode (speech recognition + text-to-speech).

# onnxruntime-gpu ships both CPU and GPU providers.
# The CUDA provider DLL links the full CUDA 12 runtime stack, so cuDNN/cuBLAS
# alone are NOT enough — it also needs cudart, cuFFT and cuRAND. These wheels
# supply every DLL so no system CUDA toolkit is required.
# On machines without a CUDA-capable GPU the wheels install but are never loaded;
# the app auto-selects CPUExecutionProvider at runtime.
onnxruntime-gpu ; platform_system != "Darwin"
onnxruntime ; platform_system == "Darwin"
nvidia-cuda-runtime-cu12>=12.0,<13 ; platform_system == "Windows"
nvidia-cudnn-cu12>=9.0,<10 ; platform_system == "Windows"
nvidia-cublas-cu12>=12.0,<13 ; platform_system == "Windows"
nvidia-cufft-cu12>=11.0,<12 ; platform_system == "Windows"
nvidia-curand-cu12>=10.0,<11 ; platform_system == "Windows"
